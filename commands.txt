# Navigate to project directory
cd project/

#build project and make executable into build folder
g++ -std=gnu++11 -O2 src/main.cpp -I./src -lgdi32 -o build/imageProcessor.exe

# Adjust brightness (increase by 50)
./build/imageProcessor --brightness 50 images/input.bmp output_images/bright_output.bmp

# Adjust contrast (factor 1.5)  
./build/imageProcessor --contrast 1.5 images/input.bmp output_images/contrast_output.bmp

# Convert to grayscale
./build/imageProcessor --grayscale images/input.bmp output_images/gray_output.bmp

# Show help
./build/imageProcessor
